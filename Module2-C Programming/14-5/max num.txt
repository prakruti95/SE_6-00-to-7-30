#include<stdio.h>
#include<conio.h>
int main()
{
    int n = 10; // Fixed to 10 elements
    int i;
    int a[50];
    int pos = 0;
    int neg = 0;
    int max; // Variable to store the maximum number

    printf("Enter 10 elements of the array:\n");
    for(i = 0; i < n; i++)
    {
        printf("Element %d: ", i + 1);
        scanf("%d", &a[i]);
    }

    max = a[0]; // Initialize max with the first element

    for(i = 0; i < n; i++)
    {
        if(a[i] > max)
        {
            max = a[i]; // Update max if a larger number is found
        }

       
    }

   
    printf("\nMaximum Element: %d", max); // Print the maximum number

    return 0;
}
